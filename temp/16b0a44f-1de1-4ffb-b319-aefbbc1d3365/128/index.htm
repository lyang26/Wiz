public HashMap<String, String> checkStartEndTime(DateTimeVo inTime,List<DateTimeVo> timeList) throws Exception{
        HashMap<String, String> result = new HashMap<String, String>();
        result.put("message", "success");
        long iStartDateMiles = inTime.getStartTime();
        long iEndDateMiles = inTime.getEndTime();
        String Mes="";
        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss"); 
        //循环查询出的时间
        for(int i=0;i<timeList.size();i++){
            DateTimeVo timeObj = timeList.get(i);
            
            long SstartDateMiles=timeObj.getStartTime();
            long SendDateMiles=timeObj.getEndTime();
            
            String strat = sdf.format(timeObj.getStartDate());
            String end = sdf.format(timeObj.getEndDate());
            //判断输入时间与查询时间的关系
            if(SstartDateMiles>=iStartDateMiles && SendDateMiles<=iEndDateMiles){
                Mes="error";
                result.put("startDate", strat);
                result.put("endDate", end);
                result.put("message", Mes);
                break;
            }else if(SstartDateMiles<=iStartDateMiles && iEndDateMiles <=SendDateMiles){
                Mes="error";
                result.put("startDate", strat);
                result.put("endDate", end);
                result.put("message", Mes);
                break;
            }else if(SstartDateMiles<= iStartDateMiles && SendDateMiles>=iStartDateMiles && iEndDateMiles>=SendDateMiles){
                Mes="error";
                result.put("startDate", strat);
                result.put("endDate", end);
                result.put("message", Mes);
                break;
            }else if(SstartDateMiles<=iEndDateMiles && SendDateMiles>=iEndDateMiles && iStartDateMiles <=SstartDateMiles){
                Mes="error";
                result.put("startDate", strat);
                result.put("endDate", end);
                result.put("message", Mes);
                break;
            }
            
        }
        return result;
    }
