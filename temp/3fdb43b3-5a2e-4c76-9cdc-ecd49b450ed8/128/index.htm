function compileTime(){
        var data = form.getData(false,true);
        var json =nui.encode(data);
        var flag = false;
        $.ajax({
            url:"<%=request.getContextPath()%>/facility/compileTime",
            type:'POST',
            data:json,
            cache:false,
            async:false,
            contentType:'application/json',
            success:function(text){
                var result = text.isSuccess;
                if(result == false){
                    flag =false;
                }else{
                    flag =true;
                }
            }
        });
        return flag;
    }
