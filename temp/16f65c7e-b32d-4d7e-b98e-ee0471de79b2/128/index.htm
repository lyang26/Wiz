检测是否为Double:
function checkDoubleData(e){
        var reg=/^\d+(\.\d+)?$/;
        if(!reg.test(this.getValue())){
             nui.alert("请填写数字类型","温馨提示");
             this.setValue("");
        }else{
            var newMoney = Number(this.getValue());
                this.setValue(newMoney);
        }
    }  

检测号码
 function checkTelData(e){
               var reg =/^1(3|4|5|7|8)\d{9}$/;
            if(!reg.test(this.getValue()) ||  this.getValue().length!=11){
                 nui.alert("请填正确11位手机号码","温馨提示");
                 this.setValue("");
            }else{
                this.setValue(this.getValue());
            }
          }
