--人员管理
CREATE TABLE `crm_hr_employ_info` (
  `hr_id` varchar(15) NOT NULL DEFAULT '' COMMENT '推荐表单号',
  `service_region` varchar(20) DEFAULT NULL COMMENT '定岗服务范围',
  `sure_identity` varchar(20) DEFAULT NULL COMMENT '确认身份',
  `plan_entry` date DEFAULT NULL COMMENT '计划入职日期',
  `wage_date` date DEFAULT NULL COMMENT '定薪日期',
  `wage_level` varchar(10) DEFAULT NULL COMMENT '薪资级别',
  `wage_base` double(10,2) DEFAULT NULL COMMENT '基本工资标准',
  `wage_work` double(10,2) DEFAULT NULL COMMENT '出勤工资标准',
  `wage_reward` double(10,2) DEFAULT NULL COMMENT '目标完成奖标准',
  `valid_date` date DEFAULT NULL COMMENT '生效日期',
  `contract_id` varchar(50) DEFAULT NULL COMMENT '合同编码',
  `contract_sign_date` date DEFAULT NULL COMMENT '合同签订日期',
  `contract_start_date` date DEFAULT NULL COMMENT '合同开始日期',
  `contract_party_a` varchar(50) DEFAULT NULL COMMENT '合同甲方',
  `contract_party_b` varchar(50) DEFAULT NULL COMMENT '合作乙方',
  `party_b_emp` varchar(50) DEFAULT NULL COMMENT '乙方工作人员',
  `work_province` varchar(15) DEFAULT NULL COMMENT '合同工作省份',
  `work_province_name` varchar(50) DEFAULT NULL COMMENT '合同工作省份名称',
  `work_city` varchar(15) DEFAULT NULL COMMENT '合同工作城市',
  `work_city_name` varchar(50) DEFAULT NULL COMMENT '合同工作城市名称',
  `live_province_code` varchar(15) DEFAULT NULL COMMENT '省份编码',
  `live_province_name` varchar(50) DEFAULT NULL COMMENT '省份名称',
  `live_city_code` varchar(15) DEFAULT NULL COMMENT '城市编码',
  `live_city_name` varchar(50) DEFAULT NULL COMMENT '城市名称',
  `live_county_code` varchar(15) DEFAULT NULL COMMENT '区县编码',
  `live_county_name` varchar(50) DEFAULT NULL COMMENT '区县名称',
  `live_street_code` varchar(15) DEFAULT NULL COMMENT '乡镇街道编码',
  `live_street_name` varchar(50) DEFAULT NULL COMMENT '乡镇街道名称',
  `live_village_code` varchar(50) DEFAULT NULL COMMENT '社区村编码',
  `live_village_name` varchar(50) DEFAULT NULL COMMENT '社区村名称',
  `live_address` varchar(255) DEFAULT NULL COMMENT '详细地址',
  `change_code` varchar(30) DEFAULT NULL COMMENT '修改人编码',
  `change_name` varchar(50) DEFAULT NULL COMMENT '修改人姓名',
  `change_date` datetime DEFAULT NULL COMMENT '修改时间',
  `acual_entry_date` datetime DEFAULT NULL COMMENT '实际入职时间',
  `entry_file1` varchar(100) DEFAULT NULL COMMENT '文件名1',
  `file_path1` varchar(255) DEFAULT NULL COMMENT '文件1路径',
  `file_path2` varchar(255) DEFAULT NULL COMMENT '文件2路径',
  `entry_file2` varchar(100) DEFAULT NULL COMMENT '文件2',
  `file_path3` varchar(255) DEFAULT NULL COMMENT '文件3路径',
  `entry_file3` varchar(100) DEFAULT NULL COMMENT '文件3',
  PRIMARY KEY (`hr_id`),
  KEY `index_identity` (`sure_identity`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='定岗定薪人员入职表';



CREATE TABLE `crm_hr_employ_store` (
  `store_id` int(8) NOT NULL AUTO_INCREMENT COMMENT '门店id',
  `hr_id` varchar(15) DEFAULT NULL COMMENT '推荐表单号',
  `store_code` varchar(20) DEFAULT NULL COMMENT '门店编码',
  `store_name` varchar(100) DEFAULT NULL COMMENT '门店名称',
  `province_code` varchar(15) DEFAULT NULL COMMENT '省份编码',
  `province_name` varchar(50) DEFAULT NULL COMMENT '省份名称',
  `city_code` varchar(15) DEFAULT NULL COMMENT '城市编码',
  `city_name` varchar(50) DEFAULT NULL COMMENT '城市名称',
  `county_code` varchar(15) DEFAULT NULL COMMENT '区县编码',
  `county_name` varchar(50) DEFAULT NULL COMMENT '区县名称',
  `street_code` varchar(15) DEFAULT NULL COMMENT '乡镇街道编码',
  `street_name` varchar(50) DEFAULT NULL COMMENT '乡镇街道名称',
  `village_code` varchar(50) DEFAULT NULL COMMENT '社区村编码',
  `village_name` varchar(50) DEFAULT NULL COMMENT '社区村名称',
  `address` varchar(255) DEFAULT NULL COMMENT '详细地址',
  PRIMARY KEY (`store_id`),
  KEY `index_hr` (`hr_id`)
) ENGINE=InnoDB AUTO_INCREMENT=24 DEFAULT CHARSET=utf8 COMMENT='人员定岗时选择的门店信息';

CREATE TABLE `crm_hr_evaluation_class` (
  `evaluate_id` int(8) NOT NULL AUTO_INCREMENT COMMENT '评价id',
  `hr_id` varchar(15) DEFAULT NULL COMMENT '推荐表单号',
  `class_type` varchar(10) DEFAULT NULL COMMENT '课程类型',
  `class_name` varchar(20) DEFAULT NULL COMMENT '培训课程',
  `class_score` double(10,2) DEFAULT NULL COMMENT '分数',
  `class_evaluate` varchar(150) DEFAULT NULL COMMENT '评价',
  `evaluate_code` varchar(20) DEFAULT NULL COMMENT '评价人',
  `evaluate_name` varchar(50) DEFAULT NULL COMMENT '评价人姓名',
  PRIMARY KEY (`evaluate_id`),
  KEY `index_hr` (`hr_id`)
) ENGINE=InnoDB AUTO_INCREMENT=74 DEFAULT CHARSET=utf8 COMMENT='人员评价课程评价表';


CREATE TABLE `crm_hr_evaluation_info` (
  `hr_id` varchar(15) NOT NULL DEFAULT '' COMMENT '推荐单号',
  `pj_plan_date` date DEFAULT NULL COMMENT '普及课计划参加日期',
  `pj_actual_date` date DEFAULT NULL COMMENT '普及课实际参加日期',
  `pj_total` double(10,1) DEFAULT NULL COMMENT '普及课总分',
  `pj_average` double(10,1) DEFAULT NULL COMMENT '普及课平均分',
  `pj_evaluate` varchar(255) DEFAULT NULL COMMENT '普及课综合评价',
  `pj_direction` varchar(155) DEFAULT NULL COMMENT '普及课人员去向',
  `sd_plan_date` date DEFAULT NULL COMMENT '深度培训计划参加日期',
  `pj_status` varchar(32) DEFAULT NULL COMMENT '是否普及评价状态：done已评价， wait：待评价',
  `sd_actual_date` date DEFAULT NULL COMMENT '深度培训实际参加日期',
  `sd_total` double(10,1) DEFAULT NULL COMMENT '深度培训总分',
  `sd_average` double(10,1) DEFAULT NULL COMMENT '深度培训平均分',
  `sd_evaluate` varchar(255) DEFAULT NULL COMMENT '深度培训综合评价',
  `sd_direction` varchar(155) DEFAULT NULL COMMENT '深度培训人员去向',
  `sg_performance` int(8) DEFAULT NULL COMMENT '试岗业绩（套）',
  `sd_status` varchar(32) DEFAULT NULL COMMENT '是否深度评价状态：done已评价， wait：待评价',
  `sg_lecturer` varchar(20) DEFAULT NULL COMMENT '试岗讲师编码',
  `sg_lecturer_name` varchar(50) DEFAULT NULL COMMENT '试岗讲师姓名',
  `sg_evaluate` varchar(255) DEFAULT NULL COMMENT '试岗课堂评价',
  `sg_start_date` date DEFAULT NULL COMMENT '试岗开始日期',
  `sg_end_date` date DEFAULT NULL COMMENT '试岗结束日期',
  `sg_trainer` varchar(20) DEFAULT NULL COMMENT '试岗带教人编码',
  `sg_trainer_name` varchar(50) DEFAULT NULL COMMENT '试岗带教人姓名',
  `sg_total_evaluate` varchar(255) DEFAULT NULL COMMENT '门店实践总评价',
  `sg_status` varchar(32) DEFAULT NULL COMMENT '是否试岗评价状态：done已评价， wait：待评价',
  `jd_result` varchar(10) DEFAULT NULL COMMENT '人员鉴定结果 pass-通过 failure-不通过',
  `jd_level` varchar(10) DEFAULT NULL COMMENT '人员鉴定评价等级',
  `jd_region` varchar(20) DEFAULT NULL COMMENT '人员鉴定推荐服务范围',
  `jd_identity` varchar(20) DEFAULT NULL COMMENT '人员鉴定推荐身份',
  `jd_recom_level` varchar(10) DEFAULT NULL COMMENT '人员鉴定推荐级别',
  `jd_in_date` date DEFAULT NULL COMMENT '人员鉴定推荐入职日期',
  `jd_evaluate` varchar(255) DEFAULT NULL COMMENT '人员鉴定综合评价',
  `change_code` varchar(32) DEFAULT NULL COMMENT '修改编码',
  `change_name` varchar(32) DEFAULT NULL COMMENT '修改人名称',
  `change_time` datetime DEFAULT NULL COMMENT '修改时间',
  PRIMARY KEY (`hr_id`),
  KEY `index_identity` (`jd_identity`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='人员评价评价信息记录表';


SELECT
`chp`.`bp_code` AS `bp_code`,
`chp`.`person_name` AS `person_name`,
`chp`.`service_region` AS `service_region`,
`chp`.`identity_id` AS `identity_id`,
`cpi`.`identity_name` AS `identity_name`,
`chp`.`person_status` AS `person_status`,
`chei`.`plan_entry` AS `plan_entry`,
`chei`.`contract_sign_date` AS `contract_sign_date`,
`chei`.`contract_start_date` AS `contract_start_date`,
`chei`.`acual_entry_date` AS `acual_entry_date`,
`chp`.`person_mobileno` AS `person_mobileno`,
`chp`.`person_idcard` AS `person_idcard`,
`chp`.`work_province` AS `work_province`,
`chp`.`work_city` AS `work_city`,
`chp`.`work_province_name` AS `work_province_name`,
`chp`.`work_city_name` AS `work_city_name`,
`chp`.`create_time` AS `create_time`,
`chei`.`change_date` AS `change_date`,
`chp`.`hr_id` AS `hr_id`
FROM
(
(
`crm_hr_person_apply` `chp`
JOIN `crm_hr_employ_info` `chei` ON (
(
`chei`.`hr_id` = `chp`.`hr_id`
)
)
)
JOIN `crm_partner_identity` `cpi` ON (
(
(
`chp`.`identity_id` = `cpi`.`identity_id`
)
AND (
`chp`.`person_status` IN (6, 7, 8)
)
)
)
)



SELECT
`chp`.`bp_code` AS `bp_code`,
`chp`.`person_name` AS `person_name`,
`chp`.`service_region` AS `service_region`,
`chp`.`identity_id` AS `identity_id`,
`cpi`.`identity_name` AS `identity_name`,
`chp`.`person_status` AS `person_status`,
`chp`.`person_mobileno` AS `person_mobileno`,
`chp`.`person_idcard` AS `person_idcard`,
`che`.`pj_status` AS `pj_status`,
`che`.`sd_status` AS `sd_status`,
`che`.`sg_status` AS `sg_status`,
`chp`.`work_province` AS `work_province`,
`chp`.`work_city` AS `work_city`,
`chp`.`work_province_name` AS `work_province_name`,
`chp`.`work_city_name` AS `work_city_name`,
`chp`.`create_time` AS `create_time`,
`chp`.`hr_id` AS `hr_id`
FROM
(
(
`crm_hr_person_apply` `chp`
LEFT JOIN `crm_hr_evaluation_info` `che` ON (
(
`chp`.`hr_id` = `che`.`hr_id`
)
)
)
JOIN `crm_partner_identity` `cpi` ON (
(
`chp`.`identity_id` = `cpi`.`identity_id`
)
)
)
WHERE
(
`chp`.`person_status` IN ('3', '4', '5', '6', '7', '8')
)



SELECT
`hpa`.`hr_id` AS `hr_id`,
`hpa`.`bp_code` AS `bp_code`,
`hpa`.`emp_id` AS `emp_id`,
`hpa`.`person_name` AS `person_name`,
`hpa`.`person_idcard` AS `person_idcard`,
`hpa`.`person_mobileno` AS `person_mobileno`,
`hpa`.`person_status` AS `person_status`,
`hpa`.`person_birthday` AS `person_birthday`,
`hpa`.`person_flag` AS `person_flag`,
`hpa`.`work_province` AS `work_province`,
`hpa`.`work_province_name` AS `work_province_name`,
`hpa`.`work_city` AS `work_city`,
`hpa`.`work_city_name` AS `work_city_name`,
`hpa`.`person_education` AS `person_education`,
`hpa`.`store_id` AS `store_id`,
`hpa`.`store_name` AS `store_name`,
`hpa`.`weixin` AS `weixin`,
`hpa`.`recommend_remark` AS `recommend_remark`,
`hpa`.`person_pic` AS `person_pic`,
`hpa`.`create_code` AS `create_code`,
`hpa`.`create_name` AS `create_name`,
`hpa`.`create_time` AS `create_time`,
`hpa`.`change_code` AS `change_code`,
`hpa`.`change_name` AS `change_name`,
`hpa`.`change_time` AS `change_time`,
`hpa`.`service_region` AS `app_region`,
`hpa`.`identity_id` AS `app_identity`,
`cpi`.`identity_name` AS `app_identity_name`,
`che`.`service_region` AS `service_region`,
`che`.`sure_identity` AS `sure_identity`,
`cpi2`.`identity_name` AS `sure_identity_name`,
`che`.`plan_entry` AS `plan_entry`,

IF (
isnull(`che`.`sure_identity`),
0,
1
) AS `posi_flag`,

IF (
isnull(`che`.`wage_level`),
0,
1
) AS `wage_flag`
FROM
(
(
(
(
`crm_hr_person_apply` `hpa`
JOIN `crm_hr_evaluation_info` `hei` ON (
(
`hei`.`hr_id` = `hpa`.`hr_id`
)
)
)
LEFT JOIN `crm_partner_identity` `cpi` ON (
(
`cpi`.`identity_id` = `hpa`.`identity_id`
)
)
)
LEFT JOIN `crm_hr_employ_info` `che` ON (
(
`che`.`hr_id` = `hpa`.`hr_id`
)
)
)
LEFT JOIN `crm_partner_identity` `cpi2` ON (
(
`cpi2`.`identity_id` = `che`.`sure_identity`
)
)
)
WHERE
(
`hpa`.`person_status` IN ('5', '6', '7', '8')
)
