public String getBpCodeZooLock(String bpName, String bpType, String typeCode,
            String inRule, int codeLength) throws Exception {
        //定义锁名，由于不同业务都调用同样的获取编码方法，使用获取主数据编码的类型来定义锁名，减少获取不到锁的几率
        if(bpType == null){
            bpType ="";
        }
        String zLockId = bpType+"Lock";
        // 此处利用Zookeeper的分布式锁来控制派单编号重复产生
        SharedLock sharedLock = new SharedLock(lockZookeeperClient, zLockId);
        String rebackCode = "";
        try {
            boolean isLock = sharedLock.acquire(200, TimeUnit.MILLISECONDS);
            log.info("========"+bpType+"产生主数据编号是否获取到锁：" + Boolean.toString(isLock));
            
            while(isLock) {
                try {
                    //获取得到锁，进行业务操作
                    Boolean isNew = false;
                    String code = "";
                    int num = 0;
                    Long codeId = 0L;
                    MallNoCodeExample example = new MallNoCodeExample();
                    MallNoCodeExample.Criteria criteria = example.createCriteria();
                    criteria.andCommandTypeEqualTo(bpType);
                    criteria.andTypeCodeEqualTo(typeCode);
                    criteria.andIncreasseRuleEqualTo(inRule);
                    List<MallNoCode> rs = mallNoCodeMapper.selectByExample(example);
                    if (rs.size() > 0) {
                        num = (Integer.valueOf(rs.get(0).getCodeNum()) + 1);
                        codeId = rs.get(0).getCodeid();
                        isNew = false;
                    } else {// 不存在编码
                        num = 1;
                        code = "1";
                        isNew = true;
                    }
                    // 规定编号的格式
                    StringBuffer sb = new StringBuffer("");
                    for (int i = 0; i < codeLength; i++) {
                        sb.append("0");
                    }
                    DecimalFormat df = new DecimalFormat(sb.toString());
                    code = df.format(num);
                    // 判断是否是新编码
                    if (isNew) {
                        MallNoCode mallNoCode = new MallNoCode();
                        mallNoCode.setCommandName(bpName);
                        mallNoCode.setTypeCode(typeCode);
                        mallNoCode.setCommandType(bpType);
                        mallNoCode.setIncreasseRule(inRule);
                        mallNoCode.setCodeNum(code);
                        mallNoCode.setLastTime(new Date());
                        mallNoCodeMapper.insertSelective(mallNoCode);
                    } else {
                        MallNoCode mallNoCode = new MallNoCode();
                        mallNoCode.setCodeid(codeId);
                        mallNoCode.setCodeNum(code);
                        mallNoCode.setLastTime(new Date());
                        mallNoCodeMapper.updateByPrimaryKeySelective(mallNoCode);
                    }
                    if (inRule == null) {
                        inRule = "";
                    }
                    rebackCode = inRule + code;
                    
                    break;
                } catch (Exception e) {
                    e.printStackTrace();
                    log.error(e.getMessage());
                }finally{
                    sharedLock.release();
                }
            }
        } catch (Exception e) {
            log.error(e.getMessage());
            e.printStackTrace();
        }
        return rebackCode;
        
    }
